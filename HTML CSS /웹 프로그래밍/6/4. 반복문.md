# 반복문

## for문

    for(초기문; 조건식; 반복 후 작업) {

        실행문

    }

ex)

    // 0에서 9까지 출력

    for(var i = 0; i < 10; i++) {

        document.write(i);

    }


## while문

    while(조건식) {

        실행문

    }

ex)

    var i = 0;

    while(i < 10) {

        document.write(i);

        i++;

    }


## do~while문

    do {

        실행문

    } while(조건식);


ex) 

    var i = 0;

    do {

        document.write(i);
        i++;


    } while(i < 10);
