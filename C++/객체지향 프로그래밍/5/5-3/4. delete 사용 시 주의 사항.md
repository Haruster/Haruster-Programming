# delete 사용 시 주의 사항

- 절적치 못한 포인터로 delete하면 실행 시간 오류가 발생한다.

    > 동적으로 할당 받지 않는 메모리 반환 - 오류

        int n;

        int *p = &n;

        delete p; // 실행 시간 오류

        // 포인터 p가 가리키는 메모리는 동적으로 할당 받은 것이 아니다.


    > 동일한 메모리 두 번 반환 - 오류

        int *p = new int;

        delete p; // 정상적인 메모리 반환

        delete p; // 실행 시간 오류. 이미 반환한 메모리를 중복으로 반환할 수 없다.
