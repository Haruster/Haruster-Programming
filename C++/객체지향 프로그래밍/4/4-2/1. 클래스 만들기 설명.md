# 클래스 만들기 설명

// 클래스 선언부

class Circle { // 클래스 이름이 Circle인 클래스 선언

    public: // 멤버에 대한 접근 지정자

        int radius; // 멤버 변수
        double getArea(); // 멤버 함수

}; // 세미콜론으로 끝난다.

// 클래스 구현부

double Circle :: getArea() { // 리턴 타입 / 이름 / 범위 지정 연산자 / 멤버 함수명과 매개 변수

    return 3.14 * radius * radius;

}
