# 2개의 리스트를 합하는 함수

    ListNode* concat_list(ListNode *head1, ListNode *head2) {

        if (head1 == NULL) return head2;

        else if (head2 == NULL) return head2;

        else {

            ListNode *p;

            p = head1;

            while(p->link != NULL) {

                p = p->link;

            }

            p->link = head2;

            return head1;

        }

    }
