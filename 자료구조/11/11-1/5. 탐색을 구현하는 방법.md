# 탐색을 구현하는 방법

1. 반복적 방법

2. 순환적 방법


## 순환적인 방법

    // 순환적인 탐색 함수

    TreeNode *search(TreeNode *node, int key) {

        if(node == NULL) return NULL;
        if(key == node->key) return node;

        else if(key < node->key) {

            return search(node->left, key);

        }

        else {

            return search(node->right, key);

        }

    }


## 반복적인 방법

    // 반복적인 탐색 함수

    TreeNode *search(TreeNode *node, int key) {

        while(node != NULL) {

            if(key == node->key) return node;

            else if(key < node->key) {

                node = node->left;

            }

            else {

                node = node->right;

            }

            return NULL;    // 탐색에 실패했을 경우 NULL을 반환한다.

        }

    }
